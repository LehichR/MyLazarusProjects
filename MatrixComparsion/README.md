function CountNegativeRows(matrix: arr; n: integer): integer;
var
  i, j: integer;
  allNegative: boolean;
begin
  Result := 0;
  for i := 0 to n - 1 do
  begin
    allNegative := true;
    for j := 0 to n - 1 do
    begin
      if matrix[i, j] >= 0 then
      begin

  end;
end;
